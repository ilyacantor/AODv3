## Step 1: Implement `validate_key_integrity`

```python
import tldextract

def validate_key_integrity(key: str) -> tuple[bool, str]:
    if not key or not key.strip():
        return False, "Empty key"
    
    key = key.strip().lower()
    
    # Reject whitespace/tabs
    if ' ' in key or '\t' in key:
        return False, "Contains whitespace"
    
    # Strict TLD Validation
    ext = tldextract.extract(key)
    if not ext.suffix:
        return False, f"No valid TLD suffix: {key}"
    
    if not ext.domain:
        return False, f"No domain component: {key}"
        
    return True, None

```

VENDOR_DOMAIN_MAP = {
    "okta": "okta.com",

    "zendesk": "zendesk.com",
    "splunk": "splunk.com",

    "slack": "slack.com"
}

def resolve_vendor_domain(vendor_name: str) -> str | None:
    normalized = vendor_name.lower().replace("-", "").replace(" ", "")
    for key, domain in VENDOR_DOMAIN_MAP.items():
        if key in normalized:
            return domain
    return None
