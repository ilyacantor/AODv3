from urllib.parse import urlparse

def clean_domain(raw_domain):
    if not raw_domain: return None
    # Strip whitespace
    domain = raw_domain.strip().lower()
    # Handle URL inputs
    if '://' in domain:
        try:
            domain = urlparse(domain).netloc
        except:
            pass
    return domain
